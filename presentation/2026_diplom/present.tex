\begin{frame}

  \frametitle{Осцилляции нейтрино в веществе}%
  %%%
  Уравнение осцилляций нейтрино в среде
  \begin{equation}\label{eq:1}
  	\imath \frac{\partial \Psi(\xi)}{\partial \xi}=H(\xi)\Psi(\xi),
    \quad \Psi(\xi)=
    \begin{pmatrix}
      \psi_1(\xi) \\
      \psi_2(\xi) \\
      \psi_3(\xi)
    \end{pmatrix},
    \quad \xi=\frac{r}{R_{\text{Sun}}}.
  \end{equation}
   \onslide<2->%
  Здесь \({H(\xi)}\) — эрмиртовая матрица
  \begin{equation*}
  	H(\xi)=H_0+\upsilon(\xi)W,
  \end{equation*}
  \onslide<3->%
  \(H_{0}\) --- гамилтониан, определяет осцилляции нейтрино в
 вакууме
	\begin{equation*}
		H_{0}=\frac{a}{E}
		\begin{pmatrix}
			0 & 0 & 0\\
			0 & b & 0\\
			0 & 0 & 1
		\end{pmatrix}.
	\end{equation*}
  \onslide<4->%
	Здесь \(E\) --- числовое значение энергии нейтрино в МэВ, а
	\(a=4,35196\times10^6\) и \(b=0,030554\) --- безразмерные параметры.

\end{frame}


\begin{frame}

  \frametitle{Осцилляции нейтрино в веществе}%
  %%%


   \onslide<1->%
  Второй член учитывает эффекты взаимодействия со средой.

  \onslide<2->%
  Матрица \(W\)
  имеет вид:
  \begin{equation*}
  	W=
  	\begin{pmatrix}
  		c_{13}^{2}c_{12}^{2} & c_{12}s_{12}c_{13}^{2} & c_{12}c_{13}s_{13}\\
  		c_{12}s_{12}c_{13}^{2} & s_{12}^{2}c_{13}^{2} & s_{12}c_{13}s_{13}\\
  		c_{12}s_{13}c_{13} & s_{12}c_{13}s_{13} & s_{13}^{2}
  	\end{pmatrix},
  \end{equation*}
  \onslide<3->%
  \(c_{ij}=\cos\theta_{ij}\), \(s_{ij}=\sin\theta_{ij}\), где \(\theta_{ij}\)
  углы смешивания. Использованные нами значения:
  \begin{equation*}
    \begin{lgathered}
      \sin^{2}\!\theta_{12}=0,308,\\
      \sin^{2}\!\theta_{23}=0,437,\\
      \sin^{2}\!\theta_{13}=0,0234.\\
    \end{lgathered}
  \end{equation*}

\end{frame}


\begin{frame}

  \frametitle{Осцилляции нейтрино в веществе}%
  %%%

  \onslide<1->%
  Рассматриваются солнечные нейтрино. Профиль плотности для солнечной модели

  \begin{equation*}
    \upsilon(\xi)=\gamma \exp^{-\eta\xi}
  \end{equation*}
  \onslide<2->%
  где \(\gamma=6,5956 \times 10^4\) и \(\eta=10,54\).

  \onslide<3->%
  Искомая физическая величина --- это вероятность выживания электронного
  нейтрино
  \begin{equation*}
    \langle P_{ee} \rangle=c_{12}^2c_{13}^2|\psi_1|^2 +
    s_{12}^2c_{13}^2|\psi_2|^2 + s_{13}^2|\psi_3|^2.
  \end{equation*}

\end{frame}

\begin{frame}
  \frametitle{Углы смешивания}%
  %%%
  Массовые состяния нейтрино  являются суперпозицией состояний флейворных
  нейтрино
  \begin{equation*}
    \ket{\nu_{i}}=\sum_{\alpha}U_{\alpha i}\ket{\nu_{i}{\alpha}}.
  \end{equation*}

  Явный вид унитарной матрицы смешивания такой:
  \begin{equation*}
    U=
    \begin{pmatrix}
      c_{12}c_{13} & s_{12}c_{13} & s_{13}\exp^{\imath\delta}\\
      -s_{12}c_{23}-c_{12}s_{23}s_{13}\exp^{\imath\delta} &
      c_{12}c_{23}-s_{12}s_{23}s_{13}\exp^{-\imath\delta} & c_{13}s_{23}\\
      s_{12}s_{23}-c_{12}c_{23}s_{13}\exp^{\imath\delta} &
      -c_{12}s_{23}-s_{12}c_{23}s_{13}\exp^{\imath\delta} & c_{13}c_{23}
    \end{pmatrix}.
  \end{equation*}
\end{frame}
